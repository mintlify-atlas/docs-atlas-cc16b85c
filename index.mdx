    Quick start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get Laravel Cashier Paystack integrated into your application in minutes
  </p>
  
  <Steps>
    <Step title="Install the package">
      Install Laravel Cashier Paystack via Composer:
      
      ```bash
      composer require veekthoven/laravel-cashier-paystack
      ```
    </Step>
    
    <Step title="Publish configuration and migrations">
      Publish the configuration file and run migrations:
      
      ```bash
      php artisan vendor:publish --provider="veekthoven\Cashier\CashierServiceProvider"
      php artisan migrate
      ```
    </Step>
    
    <Step title="Configure your Paystack keys">
      Add your Paystack API keys to your `.env` file:
      
      ```bash
      PAYSTACK_PUBLIC_KEY=pk_test_xxxxxxxxxxxxx
      PAYSTACK_SECRET_KEY=sk_test_xxxxxxxxxxxxx
      ```
    </Step>
    
    <Step title="Add the Billable trait">
      Add the `Billable` trait to your user model:
      
      ```php
      use veekthoven\Cashier\Billable;

      class User extends Authenticatable
      {
          use Billable;
      }
      ```
      
      You're ready to start managing subscriptions!
    </Step>
  </Steps>
</div>
